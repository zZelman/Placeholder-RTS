45 void CPhysicsEngine::quadTree_collision()
46 {
47  // retrieve all collidable objects
48  std::list<ARenderable*> totalObjects;
49  m_pTile_Container->getCollisiondata(&totalObjects);
50  m_pRoom_Container->getCollisiondata(&totalObjects);
51  m_pUnit_Container->getCollisiondata(&totalObjects);
52 
53  // add all objects to the quad tree
54  m_pQuadTree->clear();
55  for (std::list<ARenderable*>::iterator itr = totalObjects.begin();
56  itr != totalObjects.end();
57  ++itr)
58  {
59  m_pQuadTree->insert(*itr);
60  }
61 
62 
63  //------------------------------------
64  // Collision detection happens (below)
65  //------------------------------------
66 
67 
68  // casting pointers to try and find which object in "totalObjects" (inAll) and
69  // "returnedObjects" (inCollis) are
70  // they are allocated here because it is more efficient
71  CRoom* pRoom_inAll = NULL;
72  CUnit* pUnit_inAll = NULL;
73  CTile* pTile_inCollis = NULL;
74  CRoom* pRoom_inCollis = NULL;
75  CUnit* pUnit_inCollis = NULL;
76 
77  // bools to control wheather or not the obj has been touched
78  // by the collision resolution methods
79  bool canStepNormally_room = true;
80  bool canStepNormally_unit = true;
81 
82  // objects that the current object could collide with
83  std::list<ARenderable*> returnedObjects;
84 
85  // list of things that will not collide with anything on their next step
86  // (can update normally)
87  std::list<DPhysics*> normals;
88 
89  for (std::list<ARenderable*>::iterator all_itr = totalObjects.begin();
90  all_itr != totalObjects.end();
91  ++all_itr)
92  {
93  // the type that *all_itr is (the current pointer within returnedObjects
94  // will be non NULL, the other 2 will be NULL
95  pRoom_inAll = dynamic_cast<CRoom*>(*all_itr);
96  pUnit_inAll = dynamic_cast<CUnit*>(*all_itr);
97 
98  // apply physics to the velocities of the object
99  if (pRoom_inAll != NULL)
100  {
101  applyPhysics(pRoom_inAll);
102  }
103  else if (pUnit_inAll != NULL)
104  {
105  applyPhysics(pUnit_inAll);
106  }
107 
108  canStepNormally_room = true;
109  canStepNormally_unit = true;
110 
111  // "returnedObjects" now represents all of the objects that could potentially
112  // collide with the current ARender*
113  returnedObjects.clear();
114  m_pQuadTree->retrieve(&returnedObjects, *all_itr);
115 
116  // resolve collision detection
117  for (std::list<ARenderable*>::iterator collision_itr = returnedObjects.begin();
118  collision_itr != returnedObjects.end();
119  ++collision_itr)
120  {
121  // use casting to see which type the returned object is
122  // (returns NULL if cast was unsuccessful)
123  pTile_inCollis = dynamic_cast<CTile*>(*collision_itr);
124  pRoom_inCollis = dynamic_cast<CRoom*>(*collision_itr);
125  pUnit_inCollis = dynamic_cast<CUnit*>(*collision_itr);
126 
127  // room -> tile potential collision
128  if (pRoom_inAll != NULL && pTile_inCollis != NULL)
129  {
130  if (room_tile(pRoom_inAll, pTile_inCollis) == false)
131  {
132  canStepNormally_room = false;
133  }
134  }
135 
136  // room -> room potential collision
137  else if (pRoom_inAll != NULL && pRoom_inCollis != NULL)
138  {
139  if (room_room(pRoom_inAll, pRoom_inCollis) == false)
140  {
141  canStepNormally_room = false;
142  }
143  }
144 
145  // room -> unit potential collision
146  else if (pRoom_inAll != NULL && pUnit_inCollis != NULL)
147  {
148 
149  }
150 
151  // unit -> tile potential collision
152  else if (pUnit_inAll != NULL && pTile_inCollis != NULL)
153  {
154 
155  }
156 
157  // unit -> room potential collision
158  else if (pUnit_inAll != NULL && pRoom_inCollis != NULL)
159  {
160 
161  }
162  }
163 
164  // If you are working with [room/unit] and it has not been touched by
165  // the collision resolution methods, put it in the linked list that will
166  // be stepped normally at the end of this function
167  if (pRoom_inAll != NULL && canStepNormally_room == true)
168  {
169  normals.push_front(pRoom_inAll);
170  }
171  if (pUnit_inAll != NULL && canStepNormally_unit == true)
172  {
173  normals.push_front(pUnit_inAll);
174  }
175  }
176 
177  // all of the things that are now in "normals" are guaranteed not to hit
178  // anything in their next step, so they can step normally
179  applyStepNormally(&normals);
180 }

183 void CPhysicsEngine::n2_collision()
184 {
185  std::list<ARenderable*> rooms;
186  m_pRoom_Container->getCollisiondata(&rooms);
187 
188  std::list<ARenderable*> tiles;
189  m_pTile_Container->getCollisiondata(&tiles);
190 
191  CRoom* pRoom_from = NULL;
192  CRoom* pRoom_to = NULL;
193  CTile* pTile = NULL;
194  bool canStepNormally = true;
195  std::list<DPhysics*> normals; // objects that can step normally
196  for (std::list<ARenderable*>::iterator itr_rooms_from = rooms.begin();
197  itr_rooms_from != rooms.end();
198  ++itr_rooms_from)
199  {
200  pRoom_from = dynamic_cast<CRoom*>(*itr_rooms_from);
201  applyPhysics(pRoom_from);
202 
203  canStepNormally = true;
204 
205  //
206  // room -> tile collision detection
207  //
208  for (std::list<ARenderable*>::iterator itr_tiles = tiles.begin();
209  itr_tiles != tiles.end();
210  ++itr_tiles)
211  {
212  pTile = dynamic_cast<CTile*>(*itr_tiles);
213 
214  // returns false if the collision was resolved
215  if (room_tile(pRoom_from, pTile) == false)
216  {
217  canStepNormally = false;
218  break;
219  }
220  }
221 
222  //
223  // room -> room collision detection
224  //
225  for (std::list<ARenderable*>::iterator itr_rooms_to = rooms.begin();
226  itr_rooms_to != rooms.end();
227  ++itr_rooms_to)
228  {
229  pRoom_to = dynamic_cast<CRoom*>(*itr_rooms_to);
230 
231  // returns false if the collision was resolved
232  if (room_room(pRoom_from, pRoom_to) == false)
233  {
234  canStepNormally = false;
235  break;
236  }
237  }
238 
239  if (canStepNormally == true)
240  {
241  normals.push_front(pRoom_from);
242  }
243  }
244 
245  applyStepNormally(&normals);
246 }

249 bool CPhysicsEngine::room_tile(CRoom* pRoom, CTile* pTile)
250 {
251  // pull required data from pRoom
252  sf::FloatRect futureRect = pRoom->getGlobalBounds();
253 
254  // test the future
255  futureRect.top += pRoom->m_sPhysics.velosity_y;
256 
257  sf::FloatRect tileRect = pTile->getGlobalBounds();
258 
259  if (futureRect.intersects(tileRect))
260  {
261  int newY = tileRect.top - futureRect.height;
262  pRoom->setPosition(futureRect.left, newY);
263 
264  pRoom->m_sPhysics.gravityTimer.restart();
265 
266  pRoom->m_sPhysics.isFalling = false;
267 
268  return false;
269  }
