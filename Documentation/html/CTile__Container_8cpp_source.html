86 void CTile_Container::screenToGrid(int* posX, int* posY)
87 {
88  // x & y represent the mouse location, just need to normalize that to the grid
89  sf::Vector2<int> gridSize(m_sLayerData.width, m_sLayerData.height);
90  sf::Vector2<int> gridSubSize(m_sTilesetData.tileWidth, m_sTilesetData.tileHeight);
91 
92  // total pixels in the grid
93  int totalX = gridSize.x * gridSubSize.x;
94  int totalY = gridSize.y * gridSubSize.y;
95 
96  // compute new coords (in GRID space)
97  *posX = ((*posX) * gridSize.x) / totalX;
98  *posY = ((*posY) * gridSize.y) / totalY;

152 void CTile_Container::generateGrid()
153 {
154  CTexture* pTexture = generateGrid_texture();
155  m_pTexture = pTexture;
156 
157  // use the info received at the specific 2D index to create a sprite
158  // with the correct texture, and in the correct position
159  for (unsigned int i = 0; i < m_sLayerData.tiles.size(); ++i)
160  {
161  std::vector<int> internalVect = m_sLayerData.tiles.at(i);
162  for (unsigned int n = 0; n < internalVect.size(); ++n)
163  {
164  int data = internalVect.at(n);
165  if (data == 0) // 0 in the Tiled data structure represents "no tile present"
166  {
167  continue;
168  }
169  else
170  {
171  m_tiles.push_back(new CTile(generateGrid_sprite(n + 1, i + 1, data, pTexture)));
172  }
173  }
174  }
175 }

178 CTexture* CTile_Container::generateGrid_texture()
179 {
180  // CTexture() data fields, pulled out for readability
181  std::string fileName;
182  sf::Vector2<int> subSize;
183  sf::Vector2<int> subNum;
184 
185  // cut off everything exept the actuall file name
186  int find = m_sTilesetData.image.source.find_last_of('/');
187  fileName = m_sTilesetData.image.source.substr(find + 1, m_sTilesetData.image.source.size());
188  fileName = m_tileSetPath + fileName;
189 
190  // fill in values
191  subSize.x = m_sTilesetData.tileWidth;
192  subSize.y = m_sTilesetData.tileHeight;
193 
194  // (total image demention size) / (size of sub images) = numer of sub images
195  subNum.x = m_sTilesetData.image.width / subSize.x;
196  subNum.y = m_sTilesetData.image.height / subSize.y;
197 
198  return new CTexture(fileName, subSize, subNum);

219 void CTile_Container::findSubImage(sf::Vector2<int>* subNum, int startingLength,
220  int recievedNum, sf::Vector2<int>* returnedPos)
221 {
222 
223  // Find the row by subtracting the column size from the recievedNum
224  // you have found it when that is less than the startingLenght
225  int row = 0;
226  for (int i = 0; i < subNum->y + 1; ++i)
227  {
228  int subtract = i * subNum->x;
229  if (recievedNum - subtract < startingLength)
230  {
231  row = i;
232  break;
233  }
234  }
235 
236  // fidning the column is easy once you have the row
237  int col = recievedNum - (row - 1) * subNum->x;
238 
239 // std::cout << "(" << col << ", " << row << ")" << std::endl;
240 
241  // save the data into the return sf::Vector2
242  returnedPos->x = col;
243  returnedPos->y = row;
244 }

257 void CTile_Container::parseFile()
258 {
259  // internal flags to look for; direct children of node "map"
260  std::string tileSetNode = "tileset";
261  std::string layerNode = "layer";
262  std::string objectGroupNode = "objectgroup";
263  std::string imageLayerNode = "imagelayer";
264 
265  rapidxml::file<> file(m_filePath.c_str());
266  rapidxml::xml_document<> doc;
267  doc.parse<0>(file.data());
268 
269  //
270  // root of the DOM tree
271  //
272  rapidxml::xml_node<>* pMap = doc.first_node();
273 
274  for (rapidxml::xml_node<>* node = pMap->first_node(); node; node = node->next_sibling())
275  {
276  if (node->name() == tileSetNode)
277  {
278  parseFile_tileset(node, &m_sTilesetData);
279  }
280  else if (node->name() == layerNode)
281  {
282  parseFile_layer(node, &m_sLayerData);
283  }
284  else if (node->name() == objectGroupNode)
285  {
286  parseFile_objectgroup(node, &m_sObjectgroupData);
287  }
288  }
289 }
