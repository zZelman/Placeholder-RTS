156 int CQuadTree::getIndex(ARenderable* pObject)
157 {
158  int index = IN_PARENT;
159  double verticalMidpoint = m_bounds.left + (m_bounds.width / 2);
160  double horizontalMidpoint = m_bounds.top + (m_bounds.height / 2);
161 
162  sf::FloatRect rect = pObject->getSprite_API()->getGlobalBounds();
163 
164  // pObject can completely fit within the top quadrants
165  bool topQuadrant = (rect.top < horizontalMidpoint) && (rect.top + rect.height < horizontalMidpoint);
166  // pObject can completely fit within the bottom quadrents
167  bool bottomQuadrant = rect.top > horizontalMidpoint;
168 
169  // pObject can completely fit within the left quadrants
170  if ((rect.left < verticalMidpoint) && (rect.left + rect.width < verticalMidpoint))
171  {
172  if (topQuadrant)
173  {
174  index = 1;
175  }
176  else if (bottomQuadrant)
177  {
178  index = 2;
179  }
180  }
181  // pObject can completely fit within the right quadrants
182  else if (rect.left > verticalMidpoint)
183  {
184  if (topQuadrant)
185  {
186  index = 0;
187  }
188  else if (bottomQuadrant)
189  {
190  index = 3;
191  }
192  }
193 
194  return index;
