319 bool CRoom_Container::canSpawnRoom()
320 {
321  // mouse click (keyPressState describes whether or not it is on or off)
322  if (m_sSpawnInfo.isMouse == true)
323  {
324  int x = m_sSpawnInfo.pEvent->mouseButton.x;
325  int y = m_sSpawnInfo.pEvent->mouseButton.y;
326 
327  // click needs to be not in a room, and above a
328  // room that is on the same x coords to pass
329  if (canSpawnAtPos(x, y) == false)
330  {
331  return false;
332  }
333 
334  // prep the x,y of the mouse for room spawn
335  normalizeToGrid(&x, &y);
336 
337  // mouse was pressed
338  if (m_sSpawnInfo.keyPressState == true
339  && *(m_sSpawnInfo.spawnKeyState) == true)
340  {
341  // call the specified spawner function by resolving scope to this object,
342  // then derefferenceing the function pointer, and then passing the arguments
343  (this->*m_sSpawnInfo.spawnFunction)(x, y);
344 
345  return true;
346  }
347  // Mouse released
348  else if (m_sSpawnInfo.keyPressState == false
349  && *(m_sSpawnInfo.spawnKeyState) == true)
350  {
351 
352  }
353  }
354  else // key click
355  {
356  // key pressed
357  if (m_sSpawnInfo.keyPressState == true
358  && m_sSpawnInfo.pEvent->key.code == *(m_sSpawnInfo.spawnKey))
359  {
360  *(m_sSpawnInfo.spawnKeyState) = true;
361  return true;
362  }
363  // key released
364  else if (m_sSpawnInfo.keyPressState == false
365  && m_sSpawnInfo.pEvent->key.code == *(m_sSpawnInfo.spawnKey))
366  {
367  *(m_sSpawnInfo.spawnKeyState) = false;
368  return true;
369  }
370  }
371 
372  // Criterion not met to spawn or update the spawn conditions for a room
373  return false;
374 }

479 bool CRoom_Container::canSpawnAtPos(int x, int y)
480 {
481  sf::FloatRect rect;
482  CRoom* pR;
483 
484  // defined region of space for checking (of the current room)
485  int Rx_min = 0;
486  int Rx_max = 0;
487  int Ry = 0;
488 
489  for (std::list<CRoom*>::iterator itr = m_rooms.begin();
490  itr != m_rooms.end();
491  ++itr)
492  {
493  pR = (*itr);
494  rect = pR->getGlobalBounds();
495 
496  Rx_min = rect.left;
497  Rx_max = rect.left + rect.width;
498  Ry = rect.top;
499 
500  // check to see if mouse is within a room
501  if (rect.contains(x, y))
502  {
503  return false;
504  }
505 
506  // check to see if mouse is above a room that is below the mouse
507  if (y >= Ry && (Rx_min <= x && x <= Rx_max))
508  {
509  return false;
510  }
511  }
512 
513  return true;
514 }
