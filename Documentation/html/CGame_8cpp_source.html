124 void CGame::gameLoop()
125 {
126  isRunning = true;
127 
128  while (m_pWindow->isOpen() && isRunning)
129  {
130 
131  // NOTE: events must be polled within the main window thread
132  sf::Event event;
133  while (m_pWindow->pollEvent(event))
134  {
135  // window was closed by user
136  if (event.type == sf::Event::Closed)
137  {
138  m_pWindow->close();
139  return;
140  }
141 
142  // update game system information (size, focus, etc)
143  bool stateConsumed = input_gameSystem(&event);
144 
145  // only allow the user to supply input to the game when not paused
146  // and the input was not a system call
147  if (!isPaused && !stateConsumed)
148  {
149  input_user(&event);
150  }
151  }
152 
153  if (!isPaused) // only update states if game is actively running
154  {
155  update(); // update all game states
156  }
157 
158  render(); // render all game things based on their states
159  }
160 
161  // if the program flow gets here, its because isRunning was requested as being false
162  // therefore, you still need to close the window
163  m_pWindow->close();
164 }
